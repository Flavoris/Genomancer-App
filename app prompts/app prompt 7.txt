import Foundation

public func iupacSet(_ c: Character) -> Set<Character> { IUPAC.map[c] ?? [] }

public func complementSet(_ s: Set<Character>) -> Set<Character> {
    let comp: [Character:Character] = ["A":"T","T":"A","C":"G","G":"C"]
    return Set(s.compactMap { comp[$0] })
}

public func compatible(a5to3: String, b5to3: String) -> Bool {
    let A = Array(a5to3.uppercased()), B = Array(b5to3.uppercased())
    guard A.count == B.count else { return false }
    for i in 0..<A.count {
        let sA = iupacSet(A[i])
        let sB = complementSet(iupacSet(B[i]))
        if sA.intersection(sB).isEmpty { return false }
    }
    return true
}
